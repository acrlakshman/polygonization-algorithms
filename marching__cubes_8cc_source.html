61 template <typename T>
62 T SCALAR_POLYGONIZATION::MarchingCubes<T>::edgeIntersectionWeight(const T alpha1, const T alpha2,
63  const T iso_alpha) const
64 {
65  if (fabs(iso_alpha - alpha1) < 1e-5) {
66  return 0;
67  }
68  if (fabs(iso_alpha - alpha2) < 1e-5) {
69  return 1;
70  }
71  if (fabs(alpha1 - alpha2) < 1e-5) {
72  return 0;
73  }
74  return (iso_alpha - alpha1) / (alpha2 - alpha1);
75 }

77 template <typename T>
78 SCALAR_POLYGONIZATION::TriangleVertexTuple_t<T> SCALAR_POLYGONIZATION::MarchingCubes<T>::marchCube(
79  const std::vector<Vec3<T>>& cube_vertices, const std::vector<size_t>& edge_ids, const std::vector<T>& scalars,
80  const std::vector<Vec3<T>>& normals, const T iso_alpha)
81 {
82  std::vector<Triangle<T>> triangles;
83  std::vector<Vertex<T>> triangle_vertices;
84 
85  // Find which cube_vertices are inside of the surface and which are outside.
86  int vertex_flag = 0;
87  for (int i = 0; i < 8; ++i)
88  if (scalars[i] < iso_alpha) vertex_flag |= (1 << i);
89 
90  // If the cube is entirely inside or outside of the surface, then there will be no intersections.
91  if (edge_table[vertex_flag] == 0) {
92  TriangleVertexTuple_t<T> triangle_vertex_tuple = std::make_tuple(triangles, triangle_vertices);
93  return triangle_vertex_tuple;
94  }
95 
96  // Find the point of intersection of the surface with each edge. Then find the normal to the surface at those points.
97  Vec3<T> vertex_on_edge[12];
98  Vec3<T> normal_at_vertex_on_edge[12];
99  auto frac = static_cast<T>(0.);
100 
101  for (int edge = 0; edge < 12; ++edge) {
102  if (edge_table[vertex_flag] & (1 << edge)) {
103  frac =
104  this->edgeIntersectionWeight(scalars[edge_connection[edge][0]], scalars[edge_connection[edge][1]], iso_alpha);
105  vertex_on_edge[edge] = cube_vertices[edge_connection[edge][0]] * (static_cast<T>(1.) - frac) +
106  cube_vertices[edge_connection[edge][1]] * frac;
107  normal_at_vertex_on_edge[edge] =
108  normals[edge_connection[edge][0]] * (static_cast<T>(1.) - frac) + normals[edge_connection[edge][1]] * frac;
109  }
110  }
111 
112  // Create triangles.
113  for (int i_tri = 0; triangle_table[vertex_flag][i_tri] != -1; i_tri += 3) {
114  Triangle<T> triangle;
115  for (int i_vert = 0; i_vert < 3; ++i_vert) {
116  Vertex<T> vertex;
117  auto vertex_idx = triangle_table[vertex_flag][i_tri + i_vert];
118  vertex.pos = vertex_on_edge[vertex_idx];
119  vertex.normal = normal_at_vertex_on_edge[vertex_idx];
120  vertex.id = edge_ids[vertex_idx]; // id of the edge.
121 
122  triangle.vertex_ids[i_vert] = vertex.id;
123  triangle.normal = triangle.normal + vertex.normal;
124 
125  triangle_vertices.push_back(std::move(vertex));
126  }
127  triangle.normal = triangle.normal * static_cast<T>(SCALAR_POLYGONIZATION::one_third);
128 
129  triangles.push_back(std::move(triangle));
130  }
131 
132  TriangleVertexTuple_t<T> triangle_vertex_tuple = std::make_tuple(std::move(triangles), std::move(triangle_vertices));
133 
134  return triangle_vertex_tuple;
135 }
